#include <bits/stdc++.h>
#include <vector>
using namespace std;

class Node
{
public:
  int val = 0;
  Node* left = nullptr;
  Node* right = nullptr;

  Node(int val)
  {
    this->val = val;
  }
};

void putLeftNodesToStack(Node* root, stack<Node*> &st)
{
    while(root != NULL)
    {
        st.push(root);
        root = root->left;
    }
}

Node* bToDLL(Node* root)
{
  stack<Node*> st;
  
  // first make a dummy node
  Node* dummy = new Node(-1);
  Node *prev = dummy;
  
  // put all the left subtree nodes of root to stack
  putLeftNodesToStack(root, st);
  
  while(!st.empty())
  {
      Node *curr = st.top();
      st.pop();
      
      // doing main work
      prev->right = curr;
      curr->left = prev;
      prev = curr;
      
      putLeftNodesToStack(curr->right, st);     // putting all left nodes of right subtree of current node
  }
  
  // make root as right of dummy
  root = dummy->right;
  
  dummy->right = root->left = NULL;
  
  root->left = prev;
  prev->right = root;
  
  return root;
}

void display(Node* node)
{
  Node* head = node;
  while (node != nullptr) {
    cout << node->val << " ";
    node = node->right;
    if (node == head)
      break;
  }
}

Node* constructFromInOrder(vector<int>& inOrder, int si, int ei)
{
  if (si > ei)
    return nullptr;
  int mid = (si + ei) / 2;
  Node* root = new Node(inOrder[mid]);

  root->left = constructFromInOrder(inOrder, si, mid - 1);
  root->right = constructFromInOrder(inOrder, mid + 1, ei);

  return root;
}

Node* constructFromInOrder(vector<int>& inOrder)
{
  return constructFromInOrder(inOrder, 0, inOrder.size() - 1);
}

void solve()
{
  int n;
  cin >> n;
  vector<int> in(n, 0);
  for (int i = 0; i < n; i++)
  {
    cin >> in[i];
  }

  Node* root = constructFromInOrder(in);
  Node* head = bToDLL(root);
  display(head);
}

int main()
{
  solve();
  return 0;
}
